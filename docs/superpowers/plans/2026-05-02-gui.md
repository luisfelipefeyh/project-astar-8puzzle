```java
package puzzle.search;

import puzzle.model.Board;
import puzzle.model.Move;
import java.util.List;

public final class StepSnapshot {

    public final int stepIndex;
    public final Board board;
    public final Move move;   // null para o step 0 (estado inicial)
    public final int g, h, f;
    public final List<FrontierEntry> frontierTop;
    public final int visitedCount;
    public final String logEntry;

    public static final class FrontierEntry {
        public final Board board;
        public final int g, h, f;
        public FrontierEntry(Board board, int g, int h, int f) {
            this.board = board; this.g = g; this.h = h; this.f = f;
        }
    }

    public StepSnapshot(int stepIndex, Board board, Move move, int g, int h,
                        List<FrontierEntry> frontierTop, int visitedCount, String logEntry) {
        this.stepIndex    = stepIndex;
        this.board        = board;
        this.move         = move;
        this.g            = g;
        this.h            = h;
        this.f            = g + h;
        this.frontierTop  = frontierTop;
        this.visitedCount = visitedCount;
        this.logEntry     = logEntry;
    }
}
```

## Task 2: TracingSearch

**Files:**

- Create: `src/main/java/puzzle/search/TracingSearch.java`

TracingSearch duplica a lógica do BestFirstSearch adicionando: (1) log de decisões no terminal para CADA expansão, (2) construção de `steps[]` para o caminho solução após o search.

- [ ] **Step 1: Create TracingSearch.java**

```java
package puzzle.search;

import puzzle.heuristic.Heuristic;
import puzzle.model.Board;
import puzzle.model.Move;
import puzzle.model.Node;

import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.PriorityQueue;

public final class TracingSearch {

    private final Heuristic heuristic;
    private final String label;

    public TracingSearch(Heuristic heuristic, String label) {
        this.heuristic = heuristic;
        this.label     = label;
    }

    public TracingResult solve(Board start) {
        if (!start.isSolvable()) {
            System.out.println("[" + label + "] Tabuleiro não solucionável.");
            return new TracingResult(false, 0, 0, 0L, Collections.emptyList());
        }

        long startTime = System.currentTimeMillis();
        Comparator<Node> cmp = Comparator.comparingInt(Node::f).thenComparingInt(Node::h);
        PriorityQueue<Node> frontier = new PriorityQueue<>(cmp);
        HashMap<Long, Node> openMap  = new HashMap<>();
        HashSet<Long>       closed   = new HashSet<>();

        Node root = new Node(start, null, null, 0, heuristic.estimate(start));
        frontier.add(root);
        openMap.put(start.hashKey(), root);

        int nodesVisited = 0;
        int maxFrontier  = 1;
        int expansionIdx = 0;

        while (!frontier.isEmpty()) {
            Node current = frontier.poll();
            if (current.stale) continue;

            long key = current.state().hashKey();
            openMap.remove(key);
            if (closed.contains(key)) continue;
            closed.add(key);
            nodesVisited++;
            expansionIdx++;

            // Terminal log — every expansion
            printExpansion(expansionIdx, current, frontier, closed);

            if (current.state().isGoal()) {
                long elapsed = System.currentTimeMillis() - startTime;
                System.out.printf("[%s] ✓ SOLUÇÃO em %d passos | %d nós visitados | fronteira máx. %d | %dms%n%n",
                        label, current.reconstructPath().size(), nodesVisited, maxFrontier, elapsed);

                // Build top-5 frontier entries for the drawer
                List<StepSnapshot.FrontierEntry> frontierTop = buildFrontierTop(frontier);

                // Build steps[] along the solution path
                List<StepSnapshot> steps = buildSteps(current, frontierTop, nodesVisited);

                return new TracingResult(true, nodesVisited, maxFrontier, elapsed, steps);
            }

            // Expand neighbors
            for (Move move : current.state().legalMoves()) {
                Board next  = current.state().apply(move);
                long nKey   = next.hashKey();
                if (closed.contains(nKey)) continue;

                int gNew = current.g() + 1;
                int hNew = heuristic.estimate(next);
                Node existing = openMap.get(nKey);
                if (existing != null && gNew >= existing.g()) continue;

                if (existing != null) existing.stale = true;
                Node succ = new Node(next, current, move, gNew, hNew);
                frontier.add(succ);
                openMap.put(nKey, succ);
                if (frontier.size() > maxFrontier) maxFrontier = frontier.size();
            }
        }

        long elapsed = System.currentTimeMillis() - startTime;
        System.out.printf("[%s] ✗ Sem solução | %d nós visitados | %dms%n", label, nodesVisited, elapsed);
        return new TracingResult(false, nodesVisited, maxFrontier, elapsed, Collections.emptyList());
    }

    private void printExpansion(int idx, Node n, PriorityQueue<Node> frontier, HashSet<Long> closed) {
        int[] t = n.state().tiles();
        System.out.printf("[%s] Passo %d: Expandindo g=%d h=%d f=%d%n",
                label, idx, n.g(), n.h(), n.f());
        System.out.printf("             Estado: %d %d %d / %d %d %d / %d %d %d%n",
                t[0],t[1],t[2],t[3],t[4],t[5],t[6],t[7],t[8]);

        for (Move move : n.state().legalMoves()) {
            Board next = n.state().apply(move);
            boolean visited = closed.contains(next.hashKey());
            if (visited) {
                System.out.printf("  → %-6s já visitado — descartado%n", move);
            } else {
                int gNew = n.g() + 1;
                int hNew = heuristic.estimate(next);
                int fNew = gNew + hNew;
                System.out.printf("  → %-6s g=%d h=%d f=%d%s%n",
                        move, gNew, hNew, fNew,
                        fNew <= n.f() ? "  ★ entra na fronteira" : "    entra na fronteira");
            }
        }
        System.out.println();
    }

    private List<StepSnapshot.FrontierEntry> buildFrontierTop(PriorityQueue<Node> frontier) {
        List<Node> sorted = new ArrayList<>(frontier);
        sorted.sort(Comparator.comparingInt(Node::f).thenComparingInt(Node::h));
        List<StepSnapshot.FrontierEntry> top = new ArrayList<>();
        for (int i = 0; i < Math.min(5, sorted.size()); i++) {
            Node n = sorted.get(i);
            top.add(new StepSnapshot.FrontierEntry(n.state(), n.g(), n.h(), n.f()));
        }
        return top;
    }

    private List<StepSnapshot> buildSteps(Node goalNode,
                                           List<StepSnapshot.FrontierEntry> frontierTop,
                                           int visitedCount) {
        // Reconstruct the node chain from root to goal
        List<Node> chain = new ArrayList<>();
        Node cur = goalNode;
        while (cur != null) { chain.add(cur); cur = cur.parent(); }
        Collections.reverse(chain);

        List<StepSnapshot> steps = new ArrayList<>();
        for (int i = 0; i < chain.size(); i++) {
            Node n = chain.get(i);
            String log = i == 0
                    ? "[0] Início — g=0 h=" + n.h() + " f=" + n.f()
                    : "[" + i + "] " + n.moveFromParent() + " — g=" + n.g() +
                      " h=" + n.h() + " f=" + n.f();
            steps.add(new StepSnapshot(i, n.state(), n.moveFromParent(),
                    n.g(), n.h(), frontierTop, visitedCount, log));
        }
        return steps;
    }
}
```

```java
package puzzle.ui;

import com.sun.net.httpserver.HttpExchange;
import com.sun.net.httpserver.HttpHandler;
import puzzle.heuristic.ManhattanLinearConflictHeuristic;
import puzzle.heuristic.MisplacedTilesHeuristic;
import puzzle.heuristic.OverestimateHeuristic;
import puzzle.heuristic.ZeroHeuristic;
import puzzle.model.Board;
import puzzle.search.StepSnapshot;
import puzzle.search.TracingResult;
import puzzle.search.TracingSearch;

import java.io.IOException;
import java.io.InputStream;
import java.nio.charset.StandardCharsets;
import java.util.List;

public final class SolveHandler implements HttpHandler {

    @Override
    public void handle(HttpExchange ex) throws IOException {
        // CORS preflight
        ex.getResponseHeaders().set("Access-Control-Allow-Origin", "*");
        ex.getResponseHeaders().set("Access-Control-Allow-Methods", "POST, OPTIONS");
        ex.getResponseHeaders().set("Access-Control-Allow-Headers", "Content-Type");

        if (ex.getRequestMethod().equalsIgnoreCase("OPTIONS")) {
            ex.sendResponseHeaders(204, -1);
            return;
        }
        if (!ex.getRequestMethod().equalsIgnoreCase("POST")) {
            ex.sendResponseHeaders(405, -1);
            return;
        }

        String body = readBody(ex);
        int[] tiles;
        String algo;
        try {
            tiles = extractTiles(body);
            algo  = extractAlgo(body);
        } catch (Exception e) {
            sendError(ex, 400, "Invalid request: " + e.getMessage());
            return;
        }

        TracingSearch search = buildSearch(algo);
        if (search == null) {
            sendError(ex, 400, "Unknown algo: " + algo);
            return;
        }

        Board board = new Board(tiles);
        TracingResult result = search.solve(board);

        String json = toJson(result);
        byte[] resp = json.getBytes(StandardCharsets.UTF_8);
        ex.getResponseHeaders().set("Content-Type", "application/json; charset=UTF-8");
        ex.sendResponseHeaders(200, resp.length);
        ex.getResponseBody().write(resp);
        ex.getResponseBody().close();
    }

    private static String readBody(HttpExchange ex) throws IOException {
        InputStream is = ex.getRequestBody();
        return new String(is.readAllBytes(), StandardCharsets.UTF_8);
    }

    private static int[] extractTiles(String body) {
        int start = body.indexOf('[');
        int end   = body.indexOf(']');
        String[] parts = body.substring(start + 1, end).split(",");
        int[] tiles = new int[9];
        for (int i = 0; i < 9; i++) tiles[i] = Integer.parseInt(parts[i].trim());
        return tiles;
    }

    private static String extractAlgo(String body) {
        int idx   = body.indexOf("algo");
        int start = body.indexOf('"', idx + 4) + 1;
        int end   = body.indexOf('"', start);
        return body.substring(start, end);
    }

    private static TracingSearch buildSearch(String algo) {
        switch (algo) {
            case "UCS":            return new TracingSearch(new ZeroHeuristic(),                    "UCS");
            case "NON_ADMISSIBLE": return new TracingSearch(new OverestimateHeuristic(),            "3xManhattan");
            case "MISPLACED":      return new TracingSearch(new MisplacedTilesHeuristic(),          "MisplacedTiles");
            case "MANHATTAN_LC":   return new TracingSearch(new ManhattanLinearConflictHeuristic(), "ManhattanLinearConflict");
            default: return null;
        }
    }

    private static void sendError(HttpExchange ex, int code, String msg) throws IOException {
        byte[] b = ("{\"error\":\"" + msg + "\"}").getBytes(StandardCharsets.UTF_8);
        ex.getResponseHeaders().set("Content-Type", "application/json");
        ex.sendResponseHeaders(code, b.length);
        ex.getResponseBody().write(b);
        ex.getResponseBody().close();
    }

    // ---- JSON serialization ----

    private static String toJson(TracingResult r) {
        StringBuilder sb = new StringBuilder();
        sb.append("{");
        sb.append("\"solved\":").append(r.solved).append(",");
        sb.append("\"pathLength\":").append(r.steps.isEmpty() ? 0 : r.steps.size() - 1).append(",");
        sb.append("\"nodesVisited\":").append(r.nodesVisited).append(",");
        sb.append("\"maxFrontierSize\":").append(r.maxFrontierSize).append(",");
        sb.append("\"elapsedMs\":").append(r.elapsedMs).append(",");
        sb.append("\"steps\":").append(stepsJson(r.steps));
        sb.append("}");
        return sb.toString();
    }

    private static String stepsJson(List<StepSnapshot> steps) {
        StringBuilder sb = new StringBuilder("[");
        for (int i = 0; i < steps.size(); i++) {
            if (i > 0) sb.append(",");
            StepSnapshot s = steps.get(i);
            sb.append("{");
            sb.append("\"stepIndex\":").append(s.stepIndex).append(",");
            sb.append("\"board\":").append(boardJson(s.board)).append(",");
            sb.append("\"move\":").append(s.move == null ? "null" : "\"" + s.move + "\"").append(",");
            sb.append("\"g\":").append(s.g).append(",");
            sb.append("\"h\":").append(s.h).append(",");
            sb.append("\"f\":").append(s.f).append(",");
            sb.append("\"visitedCount\":").append(s.visitedCount).append(",");
            sb.append("\"logEntry\":\"").append(escJson(s.logEntry)).append("\",");
            sb.append("\"frontierTop\":").append(frontierJson(s.frontierTop));
            sb.append("}");
        }
        return sb.append("]").toString();
    }

    private static String boardJson(puzzle.model.Board b) {
        int[] t = b.tiles();
        return "[" + t[0]+","+t[1]+","+t[2]+","+t[3]+","+t[4]+","+t[5]+","+t[6]+","+t[7]+","+t[8] + "]";
    }

    private static String frontierJson(List<StepSnapshot.FrontierEntry> fe) {
        StringBuilder sb = new StringBuilder("[");
        for (int i = 0; i < fe.size(); i++) {
            if (i > 0) sb.append(",");
            StepSnapshot.FrontierEntry e = fe.get(i);
            sb.append("{\"board\":").append(boardJson(e.board))
              .append(",\"g\":").append(e.g)
              .append(",\"h\":").append(e.h)
              .append(",\"f\":").append(e.f).append("}");
        }
        return sb.append("]").toString();
    }

    private static String escJson(String s) {
        return s.replace("\\", "\\\\").replace("\"", "\\\"");
    }
}
```

```java
package puzzle.ui;

import com.sun.net.httpserver.HttpExchange;
import com.sun.net.httpserver.HttpServer;

import java.awt.Desktop;
import java.io.IOException;
import java.net.InetSocketAddress;
import java.net.URI;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Paths;
import java.util.concurrent.Executors;

public final class PuzzleServer {

    public static void main(String[] args) throws IOException {
        int port = 8080;
        for (int i = 0; i < args.length - 1; i++) {
            if ("--port".equals(args[i])) port = Integer.parseInt(args[i + 1]);
        }

        HttpServer server = HttpServer.create(new InetSocketAddress(port), 0);
        server.createContext("/cases", new CasesHandler());
        server.createContext("/solve", new SolveHandler());
        server.createContext("/", exchange -> serveHtml(exchange));
        server.setExecutor(Executors.newFixedThreadPool(4));
        server.start();

        String url = "http://localhost:" + port;
        System.out.println("╔══════════════════════════════════════╗");
        System.out.println("║  A* 8-Puzzle GUI                     ║");
        System.out.println("║  " + url + "               ║");
        System.out.println("╚══════════════════════════════════════╝");
        System.out.println();

        if (Desktop.isDesktopSupported()) {
            try { Desktop.getDesktop().browse(new URI(url)); }
            catch (Exception ignored) {}
        }
    }

    private static void serveHtml(HttpExchange ex) throws IOException {
        if (!ex.getRequestMethod().equals("GET")) {
            ex.sendResponseHeaders(405, -1);
            return;
        }
        byte[] body = Files.readAllBytes(Paths.get("web/index.html"));
        ex.getResponseHeaders().set("Content-Type", "text/html; charset=UTF-8");
        ex.getResponseHeaders().set("Cache-Control", "no-cache");
        ex.sendResponseHeaders(200, body.length);
        ex.getResponseBody().write(body);
        ex.getResponseBody().close();
    }
}
```

```html
        document.getElementById("preset-" + i).classList.add("selected");
        state.selectedTiles = [...state.cases[i].tiles];
        checkBtnResolve();
      }

      // ══════════════════════════════════════
      //  SETUP — INTERACTIVE BOARD
      // ══════════════════════════════════════
      function renderIBoard() {
        const board = document.getElementById("iboard");
        const blank = state.interactive.indexOf(0);
        board.innerHTML = state.interactive
          .map((t, idx) => {
            if (t === 0)
              return `<div class="itile blank" data-idx="${idx}"></div>`;
            const sel = state.iSelected === idx ? " selected" : "";
            return `<div class="itile${sel}" data-idx="${idx}" onclick="clickITile(${idx})">${t}</div>`;
          })
          .join("");

        // update text
        const t = state.interactive;
        document.getElementById("iboard-text").innerHTML =
          `${t[0]} ${t[1]} ${t[2]}<br>${t[3]} ${t[4]} ${t[5]}<br>${t[6]} ${t[7]} ${t[8]}`;

        const ok = isSolvable(state.interactive);
        document.getElementById("solvable-badge").innerHTML = ok
          ? '<span class="badge badge-ok">✓ Solucionável</span>'
          : '<span class="badge badge-err">✗ Não solucionável</span>';

        if (state.tab === "interactive") {
          state.selectedTiles = ok ? [...state.interactive] : null;
          checkBtnResolve();
        }
      }

      function clickITile(idx) {
        const blank = state.interactive.indexOf(0);
        const adjacent = isAdjacent(idx, blank);
        if (adjacent) {
          // move tile into blank
          [state.interactive[blank], state.interactive[idx]] = [
            state.interactive[idx],
            state.interactive[blank],
          ];
          state.iSelected = null;
        } else {
          state.iSelected = state.iSelected === idx ? null : idx;
        }
        renderIBoard();
      }

      function isAdjacent(a, b) {
        const ra = Math.floor(a / 3),
          ca = a % 3,
          rb = Math.floor(b / 3),
          cb = b % 3;
        return Math.abs(ra - rb) + Math.abs(ca - cb) === 1;
      }

      function isSolvable(tiles) {
        let inv = 0;
        for (let i = 0; i < 9; i++) {
          if (tiles[i] === 0) continue;
          for (let j = i + 1; j < 9; j++) {
            if (tiles[j] !== 0 && tiles[i] > tiles[j]) inv++;
          }
        }
        return inv % 2 === 0;
      }

      function shuffleBoard() {
        // Shuffle from goal via random valid moves to guarantee solvability
        let tiles = [...GOAL];
        let blank = 8;
        for (let i = 0; i < 200; i++) {
          const moves = [];
          const r = Math.floor(blank / 3),
            c = blank % 3;
          if (r > 0) moves.push(blank - 3);
          if (r < 2) moves.push(blank + 3);
          if (c > 0) moves.push(blank - 1);
          if (c < 2) moves.push(blank + 1);
          const next = moves[Math.floor(Math.random() * moves.length)];
          [tiles[blank], tiles[next]] = [tiles[next], tiles[blank]];
          blank = next;
        }
        state.interactive = tiles;
        state.iSelected = null;
        renderIBoard();
      }

      function resetBoard() {
        state.interactive = [...GOAL];
        state.iSelected = null;
        renderIBoard();
      }

      // ══════════════════════════════════════
      //  SETUP — ALGO + RESOLVE
      // ══════════════════════════════════════
      function selectAlgo(btn) {
        document
          .querySelectorAll("#setup-algo-btns .algo-btn")
          .forEach((b) => b.classList.remove("active"));
        btn.classList.add("active");
        state.selectedAlgo = btn.dataset.algo;
      }

      function checkBtnResolve() {
        document.getElementById("btn-resolve").disabled = !state.selectedTiles;
      }

      async function resolve() {
        if (!state.selectedTiles) return;
        const btn = document.getElementById("btn-resolve");
        btn.textContent = "Resolvendo…";
        btn.disabled = true;

        try {
          const data = await callSolve(state.selectedTiles, state.selectedAlgo);
          state.result = data;
          state.currentStep = 0;
          state.playing = false;
          syncSolverAlgoBtns(state.selectedAlgo);
          showSolverScreen();
          renderSolverStep(0);
        } finally {
          btn.textContent = "Resolver →";
          checkBtnResolve();
        }
      }

      async function callSolve(tiles, algo) {
        const res = await fetch("/solve", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ tiles, algo }),
        });
        return await res.json();
      }

      // ══════════════════════════════════════
      //  SOLVER SCREEN
      // ══════════════════════════════════════
      function showSolverScreen() {
        document.getElementById("screen-setup").classList.add("hidden");
        document.getElementById("screen-solver").classList.remove("hidden");
      }

      function backToSetup() {
        stopPlay();
        document.getElementById("screen-solver").classList.add("hidden");
        document.getElementById("screen-setup").classList.remove("hidden");
      }

      function renderSolverStep(stepIdx) {
        const steps = state.result.steps;
        if (!steps || steps.length === 0) return;
        stepIdx = Math.max(0, Math.min(stepIdx, steps.length - 1));
        state.currentStep = stepIdx;

        const step = steps[stepIdx];
        const prevBoard = stepIdx > 0 ? steps[stepIdx - 1].board : null;

        // Board
        const grid = document.getElementById("board-grid");
        grid.innerHTML = step.board
          .map((t, i) => {
            const moved =
              prevBoard && t !== 0 && prevBoard[i] === 0 ? " moved" : "";
            if (t === 0) return `<div class="stile blank${moved}"></div>`;
            return `<div class="stile${moved}">${t}</div>`;
          })
          .join("");

        // g/h/f
        document.getElementById("ghf-row").innerHTML =
          `g=${step.g} · h=${step.h} · <strong>f=${step.f}</strong>`;

        // Progress
        const total = steps.length - 1;
        document.getElementById("step-label").textContent =
          `Passo ${stepIdx} de ${total}`;
        document.getElementById("move-label").textContent = step.move || "—";
        document.getElementById("progress-fill").style.width =
          (total === 0 ? 100 : (stepIdx / total) * 100) + "%";

        // Metrics
        document.getElementById("m-nodes").textContent =
          state.result.nodesVisited;
        document.getElementById("m-path").textContent = state.result.pathLength;
        document.getElementById("m-frontier").textContent =
          state.result.maxFrontierSize;
        document.getElementById("m-time").textContent =
          state.result.elapsedMs + "ms";

        // Drawer
        if (state.drawerOpen) renderDrawer(step, stepIdx);

        // Play button
        document.getElementById("play-btn").textContent = state.playing
          ? "⏸"
          : "▶";
      }

      // ══════════════════════════════════════
      //  PLAYBACK
      // ══════════════════════════════════════
      function goStep(n) {
        stopPlay();
        renderSolverStep(n === 999 ? state.result.steps.length - 1 : n);
      }

      function stepBy(delta) {
        stopPlay();
        renderSolverStep(state.currentStep + delta);
      }

      function togglePlay() {
        if (state.playing) {
          stopPlay();
          return;
        }
        if (state.currentStep >= state.result.steps.length - 1) {
          state.currentStep = 0;
        }
        state.playing = true;
        document.getElementById("play-btn").textContent = "⏸";
        state.playTimer = setInterval(() => {
          const next = state.currentStep + 1;
          if (next >= state.result.steps.length) {
            stopPlay();
            return;
          }
          renderSolverStep(next);
        }, state.speed);
      }

      function stopPlay() {
        state.playing = false;
        clearInterval(state.playTimer);
        document.getElementById("play-btn").textContent = "▶";
      }

      function updateSpeed(val) {
        // val 0..100 → speed 1800ms..100ms
        state.speed = Math.round(1800 - (val / 100) * 1700);
        if (state.playing) {
          stopPlay();
          togglePlay();
        }
      }

      // ══════════════════════════════════════
      //  ALGO CHANGE
      // ══════════════════════════════════════
      function syncSolverAlgoBtns(algo) {
        document
          .querySelectorAll("#solver-algo-grid .algo-card")
          .forEach((c) => {
            c.classList.toggle("active", c.dataset.algo === algo);
          });
      }

      async function changeAlgo(card) {
        const algo = card.dataset.algo;
        if (algo === state.selectedAlgo) return;
        stopPlay();
        state.selectedAlgo = algo;
        syncSolverAlgoBtns(algo);
        card.innerHTML = card.innerHTML; // force repaint
        const data = await callSolve(state.selectedTiles, algo);
        state.result = data;
        state.currentStep = 0;
        renderSolverStep(0);
      }

      // ══════════════════════════════════════
      //  COMPARE ALL
      // ══════════════════════════════════════
      async function runAll() {
        const algos = ["UCS", "NON_ADMISSIBLE", "MISPLACED", "MANHATTAN_LC"];
        const labels = {
          UCS: "UCS",
          NON_ADMISSIBLE: "3×Manhattan",
          MISPLACED: "Misplaced",
          MANHATTAN_LC: "Manhattan LC",
        };
        const tbody = document.getElementById("compare-tbody");
        tbody.innerHTML =
          '<tr><td colspan="4" style="color:#64748b;font-size:.75rem">Calculando…</td></tr>';

        const results = await Promise.all(
          algos.map((a) => callSolve(state.selectedTiles, a)),
        );
        const minNodes = Math.min(...results.map((r) => r.nodesVisited));

        tbody.innerHTML = results
          .map((r, i) => {
            const best = r.nodesVisited === minNodes ? ' class="best"' : "";
            const star = r.nodesVisited === minNodes ? " ★" : "";
            return `<tr${best}><td>${labels[algos[i]]}${star}</td>
      <td>${r.nodesVisited}</td><td>${r.pathLength}</td><td>${r.elapsedMs}ms</td></tr>`;
          })
          .join("");
      }

      // ══════════════════════════════════════
      //  DRAWER
      // ══════════════════════════════════════
      function toggleDrawer() {
        state.drawerOpen = !state.drawerOpen;
        document
          .getElementById("drawer-body")
          .classList.toggle("hidden", !state.drawerOpen);
        document.getElementById("drawer-icon").textContent = state.drawerOpen
          ? "▼"
          : "▲";
        if (state.drawerOpen && state.result) {
          renderDrawer(
            state.result.steps[state.currentStep],
            state.currentStep,
          );
        }
      }

      function renderDrawer(step, stepIdx) {
        // Frontier top-5
        const fl = document.getElementById("frontier-list");
        if (!step.frontierTop || step.frontierTop.length === 0) {
          fl.innerHTML =
            '<div style="color:#475569;font-size:.75rem">Fronteira vazia</div>';
        } else {
          fl.innerHTML = step.frontierTop
            .map((e, i) => {
              const t = e.board;
              const top = i === 0 ? " top" : "";
              return `<div class="frontier-item${top}">
        <span class="frontier-board">${t[0]}${t[1]}${t[2]}/${t[3]}${t[4] || "_"}${t[5]}/${t[6]}${t[7]}${t[8]}</span>
        <span class="frontier-ghf"><span>g=${e.g}</span><span>h=${e.h}</span><strong style="color:${i === 0 ? "#60a5fa" : "#94a3b8"}">f=${e.f}</strong></span>
      </div>`;
            })
            .join("");
        }

        // Log
        const ll = document.getElementById("log-list");
        if (!state.result.steps) return;
        ll.innerHTML = state.result.steps
          .slice(0, stepIdx + 1)
          .map((s, i) => {
            const cls = i === stepIdx ? "current" : "done";
            return `<div class="log-entry ${cls}">${s.logEntry}</div>`;
          })
          .join("");
        // scroll to bottom
        ll.scrollTop = ll.scrollHeight;
      }
    </script>
  </body>
</html>
```
